59 bool Thread::Start()
60 {
61  //
62  // If the Scheduler is on, we need to lock before checking
63  // the ThreadStarted variable. We'll leverage the LockGuard
64  // pattern, so we can create the guard and just forget it.
65  // Leaving scope, including the return, will automatically
66  // unlock it.
67  //
68  if (SchedulerActive) {
69 
70  LockGuard guard (StartGuardLock);
71 
72  if (ThreadStarted)
73  return false;
74  else
75  ThreadStarted = true;
76  }
77  //
78  // If the Scheduler isn't running, just check it.
79  //
80  else {
81 
82  if (ThreadStarted)
83  return false;
84  else
85  ThreadStarted = true;
86  }
87 
88  BaseType_t rc = xTaskCreate(TaskFunctionAdapter,
89  Name.c_str(),
90  StackDepth,
91  this,
92  Priority,
93  &handle);
94 
95  return rc != pdPASS ? false : true;
96 }
